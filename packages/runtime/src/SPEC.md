# Honeydeck Public Source Surface Specification

> Observable behavior for public imports and exported types spanning source subpackages.

## Public Imports & Types

### Import Patterns

```ts
// Core components
import { Reveal, RevealWith, RevealGroup, Fade, FadeWith, FadeGroup, TimelineSteps, ListStyle, Keyboard, BrowserFrame, Notes, ColorModeImage } from '@honeydeck/runtime'

// Types for kit authors
import type { LayoutProps, LayoutMap, LayoutDemo } from '@honeydeck/runtime/types'

// Theme CSS
import '@honeydeck/runtime/theme.css'
import '@honeydeck/runtime/themes/base.css' // alias for honeydeck/theme.css
import '@honeydeck/runtime/themes/clean.css'
import '@honeydeck/runtime/themes/bee.css'

// Individual layouts and layout-specific slot components
import BlankLayout from '@honeydeck/runtime/layouts/Blank'
import { Left, Right } from '@honeydeck/runtime/layouts/TwoCol'
import { ColorModeImage as LayoutColorModeImage } from '@honeydeck/runtime/layouts/ColorModeImage'

// Named clean layout map, layout imports, and slot components
import cleanLayouts from '@honeydeck/runtime/layouts/clean'
import CleanBlankLayout from '@honeydeck/runtime/layouts/clean/Blank'
import { Left as CleanLeft, Right as CleanRight } from '@honeydeck/runtime/layouts/clean/TwoCol'

// Bee layout map, layout imports, and slot components
import beeLayouts from '@honeydeck/runtime/layouts/bee'
import BeeBlankLayout from '@honeydeck/runtime/layouts/bee/Blank'
import { Left as BeeLeft, Right as BeeRight } from '@honeydeck/runtime/layouts/bee/TwoCol'

// Built-in placeholder image URLs for custom layouts
import {
  imagePlaceholder,
  imagePlaceholderDark,
  verticalImagePlaceholder,
  verticalImagePlaceholderDark,
} from '@honeydeck/runtime/layouts/placeholders'

// Public component barrel
import {
  Reveal as BarrelReveal,
  RevealWith as BarrelRevealWith,
  RevealGroup as BarrelRevealGroup,
  Fade as BarrelFade,
  FadeWith as BarrelFadeWith,

  Notes as BarrelNotes,
  ColorModeCycleButton,
} from '@honeydeck/runtime/components'
import type { ColorMode } from '@honeydeck/runtime/components'

// Reserved injected components (generated by Honeydeck remark plugins; not intended for manual slide author use)
import { HoneydeckCodeBlock } from '@honeydeck/runtime/components/code-block/normal'
import { HoneydeckMagicCodeBlock } from '@honeydeck/runtime/components/code-block/magic'

// Reserved app shell entry (loaded by Honeydeck CLI HTML; not intended for manual slide author use)
import '@honeydeck/runtime/app-shell'
```

### Type Definitions

```ts
import type { ComponentType, ReactNode } from 'react'

/**
 * Props passed to layout components.
 * Generic parameter F types the accepted frontmatter fields.
 */
type LayoutProps<F extends Record<string, unknown> = Record<string, unknown>> = {
  title: ReactNode | null // currently a plain text string or null
  children: ReactNode
  rawChildren: ReactNode // currently same compiled content as children
  frontmatter: F
}

/**
 * A map of layout names to layout components.
 */
type LayoutMap = Record<string, ComponentType<LayoutProps<any>>>

/**
 * Demo data for the docs reference page.
 */
type LayoutDemo<F extends Record<string, unknown> = Record<string, unknown>> = {
  mdx: string
}
```
